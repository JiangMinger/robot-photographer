00117 void RPFramingNode::initializeRotationVector()
00118 {
00119     double angle_x_rad = CAMERA_ROTATION_X * M_PI / 180.0;
00120     double angle_y_rad = CAMERA_ROTATION_Y * M_PI / 180.0;
00121     double angle_z_rad = CAMERA_ROTATION_Z * M_PI / 180.0;
00122 
00123     cv::Mat rotation_x = (cv::Mat_<double>(3, 3) <<
00124             1.0,               0.0,               0.0,
00125             0.0,               cos(angle_x_rad),  -sin(angle_x_rad),
00126             0.0,               sin(angle_x_rad),  cos(angle_x_rad));
00127 
00128     cv::Mat rotation_y = (cv::Mat_<double>(3, 3) <<
00129             cos(angle_y_rad),  0.0,               sin(angle_y_rad),
00130             0.0,               1.0,               0.0,
00131             -sin(angle_y_rad), 0.0,               cos(angle_y_rad));
00132 
00133     cv::Mat rotation_z = (cv::Mat_<double>(3, 3) <<
00134             cos(angle_z_rad),  -sin(angle_z_rad), 0.0,
00135             sin(angle_z_rad),  cos(angle_z_rad),  0.0,
00136             0.0,               0.0,               1.0);
00137 
00138     cv::Mat combined_rotation = rotation_z * rotation_y * rotation_x;
00139 
00140     // Convert to Rodrigues representation (axis/angle, where angle is the magnitude of the vector)
00141     rotation_vector = cv::Mat_<double>(3, 1);
00142     cv::Rodrigues(combined_rotation, rotation_vector);
