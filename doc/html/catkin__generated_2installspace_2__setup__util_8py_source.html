00087 def _rollback_env_variable(environ, name, subfolder):
00088     '''
00089     For each catkin workspace in CMAKE_PREFIX_PATH remove the first entry from env[NAME] matching workspace + subfolder.
00090 
00091     :param subfolder: str '' or subfoldername that may start with '/'
00092     :returns: the updated value of the environment variable.
00093     '''
00094     value = environ[name] if name in environ else ''
00095     env_paths = [path for path in value.split(os.pathsep) if path]
00096     value_modified = False
00097     if subfolder:
00098         if subfolder.startswith(os.path.sep) or (os.path.altsep and subfolder.startswith(os.path.altsep)):
00099             subfolder = subfolder[1:]
00100         if subfolder.endswith(os.path.sep) or (os.path.altsep and subfolder.endswith(os.path.altsep)):
00101             subfolder = subfolder[:-1]
00102     for ws_path in _get_workspaces(environ, include_fuerte=True):
00103         path_to_find = os.path.join(ws_path, subfolder) if subfolder else ws_path
00104         path_to_remove = None
00105         for env_path in env_paths:
00106             env_path_clean = env_path[:-1] if env_path and env_path[-1] in [os.path.sep, os.path.altsep] else env_path
00107             if env_path_clean == path_to_find:
00108                 path_to_remove = env_path
00109                 break
00110         if path_to_remove:
00111             env_paths.remove(path_to_remove)
00112             value_modified = True
00113     new_value = os.pathsep.join(env_paths)
